void dijkstra(int a, int n) {
    for(int i = 0; i <= n; i++) d[i] = DINF; d[a] = 0;
    q.push(make_pair(0, a));
    while (!q.empty()) {
        P x = q.top(); q.pop();
        if (d[x.second] != x.first) continue; 
        for (int u = x.second, i = head[u], v; i; i = e[i].nxt) {
            v = e[i].to;
            if (d[v] < d[u] + e[i].w) continue;
                d[v] = d[u] + e[i].w;
                q.push(make_pair(d[v], v));
        }
    }
}
